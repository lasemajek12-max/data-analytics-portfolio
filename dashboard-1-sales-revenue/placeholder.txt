          <div class="visual-spec-title" style="margin-top:12px">📐 Total Revenue</div>

<span class="dax-var">Total Revenue</span> = <span class="dax-fn">SUM</span>(Sales_Transactions[Revenue_USD])

          <div class="visual-spec-title">📐 Profit Margin %</div>
          <div class="dax-block">
<span class="dax-var">Profit Margin %</span> = <span class="dax-fn">DIVIDE</span>([Total Profit], [Total Revenue], <span class="dax-num">0</span>)
          </div>

          <div class="visual-spec-title">📐 Total Deals</div>
          <div class="dax-block">
<span class="dax-var">Total Deals</span> = <span class="dax-fn">COUNTROWS</span>(Sales_Transactions)
          </div>

          <div class="visual-spec-title">📐 Avg Deal Size</div>
          <div class="dax-block">
<span class="dax-var">Avg Deal Size</span> = <span class="dax-fn">DIVIDE</span>([Total Revenue], [Total Deals], <span class="dax-num">0</span>)
          </div>

          <div class="visual-spec-title">📐 Revenue vs Target Variance</div>
          <div class="dax-block">
<span class="dax-var">Revenue vs Target</span> =
  <span class="dax-fn">SUMX</span>(
    Monthly_Targets,
    Monthly_Targets[Actual_USD] - Monthly_Targets[Target_USD]
  )
          </div>

          <div class="visual-spec-title">📐 Target Attainment %</div>
          <div class="dax-block">
<span class="dax-var">Target Attainment %</span> =
  <span class="dax-fn">DIVIDE</span>(
    <span class="dax-fn">SUM</span>(Monthly_Targets[Actual_USD]),
    <span class="dax-fn">SUM</span>(Monthly_Targets[Target_USD]),
    <span class="dax-num">0</span>
  )
          </div>

          <div class="tip-box">✅ Format currency measures: select measure → Measure tools tab → Format as <strong>Currency $</strong>, 0 decimal places</div>
